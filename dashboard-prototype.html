# Send Message Example
curl -X POST "https://api.wamate.io/v1/send" \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "instanceId": "WA_001",
    "to": "1234567890@s.whatsapp.net",
    "content": { "text": "Hello World!" }
  }'
